export const DIMENSIONS = ["technical", "impact", "cost", "team"] as const;
export type EvaluationDimension = (typeof DIMENSIONS)[number];

export type EvaluationOutput = {
  score: number;          // 0-100
  justification: string;
  recommendation: "strong_approve" | "approve" | "needs_revision" | "reject";
  keyFindings: string[];  // 1-3 items
};

export type DimensionEvaluation = {
  dimension: EvaluationDimension;
  output: EvaluationOutput;
  audit: { promptSent: string; modelUsed: string; rawResponse: string; evaluatedAt: number };
};

export type ProposalEvaluation = {
  proposalId: string;
  dimensions: DimensionEvaluation[];  // array of per-dimension results
  aggregate: { weightedScore: number; completedDimensions: number; computedAt: number };
  status: "evaluating" | "evaluated" | "failed";
};
```

];
```
