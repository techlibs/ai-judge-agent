```rust
use anchor_lang::prelude::*;

declare_id!("2RC6cF4pmnANHAPkpoR2RcPm79Zgq8G9Sz9JKotGMvS6");

#[program]
pub mod grant_evaluator {
    use super::*;

    pub fn submit_proposal(
        ctx: Context<SubmitProposal>,
        title_hash: [u8; 32],
        content_cid: [u8; 32],
        repo_url_hash: [u8; 32],
        demo_url_hash: [u8; 32],
        domain: ProposalDomain,
    ) -> Result<()> {
        require!(content_cid != [0u8; 32], ErrorCode::InvalidContentCid);

        let proposal = &mut ctx.accounts.proposal;
        let clock = Clock::get()?;

        proposal.submitter = ctx.accounts.submitter.key();
        proposal.content_cid = content_cid;
        proposal.repo_url_hash = repo_url_hash;
        proposal.demo_url_hash = demo_url_hash;
        proposal.domain = domain;
        proposal.status = ProposalStatus::Submitted;
        proposal.submitted_at = clock.unix_timestamp;
        proposal.bump = ctx.bumps.proposal;

        emit!(ProposalSubmitted {
            proposal_id: proposal.key(),
            submitter: proposal.submitter,
            content_cid,
            domain,
            submitted_at: proposal.submitted_at,
        });

        Ok(())
    }

    pub fn update_status(
        ctx: Context<UpdateStatus>,
        new_status: ProposalStatus,
    ) -> Result<()> {
        let proposal = &mut ctx.accounts.proposal;
        let current = proposal.status;

        require!(
            is_valid_transition(current, new_status),
            ErrorCode::InvalidStatusTransition
        );

        proposal.status = new_status;

        emit!(ProposalStatusChanged {
            proposal_id: proposal.key(),
            new_status,
        });

        Ok(())
    }
}

fn is_valid_transition(current: ProposalStatus, target: ProposalStatus) -> bool {
    matches!(
        (current, target),
        (ProposalStatus::Submitted, ProposalStatus::UnderReview)
            | (ProposalStatus::UnderReview, ProposalStatus::Evaluated)
            | (ProposalStatus::Evaluated, ProposalStatus::Disputed)
    )
}

#[derive(Accounts)]
#[instruction(title_hash: [u8; 32])]
pub struct SubmitProposal<'info> {
    #[account(
        init,
        payer = submitter,
        space = 8 + Proposal::INIT_SPACE,
        seeds = [b"proposal", submitter.key().as_ref(), title_hash.as_ref()],
        bump
    )]
    pub proposal: Account<'info, Proposal>,
    #[account(mut)]
    pub submitter: Signer<'info>,
    pub system_program: Program<'info, System>,
}

#[derive(Accounts)]
pub struct UpdateStatus<'info> {
    #[account(mut)]
    pub proposal: Account<'info, Proposal>,
    pub evaluator: Signer<'info>,
    // NOTE: In production, evaluator would be validated against a config account.
    // For Phase 2, any signer with the evaluator key can update status.
    // Phase 6 will add proper authority management.
}

#[account]
#[derive(InitSpace)]
pub struct Proposal {
    pub submitter: Pubkey,          // 32
    pub content_cid: [u8; 32],     // 32
    pub repo_url_hash: [u8; 32],   // 32
    pub demo_url_hash: [u8; 32],   // 32
    pub domain: ProposalDomain,     // 1
    pub status: ProposalStatus,     // 1
    pub submitted_at: i64,          // 8
    pub bump: u8,                   // 1
}

#[derive(AnchorSerialize, AnchorDeserialize, Clone, Copy, PartialEq, Eq, InitSpace)]
#[repr(u8)]
pub enum ProposalDomain {
    DeFi = 0,
    Governance = 1,
    Education = 2,
    Health = 3,
    Infrastructure = 4,
    Other = 5,
}

#[derive(AnchorSerialize, AnchorDeserialize, Clone, Copy, PartialEq, Eq, InitSpace)]
#[repr(u8)]
pub enum ProposalStatus {
    Submitted = 0,
    UnderReview = 1,
    Evaluated = 2,
    Disputed = 3,
}

#[event]
pub struct ProposalSubmitted {
    pub proposal_id: Pubkey,
    pub submitter: Pubkey,
    pub content_cid: [u8; 32],
    pub domain: ProposalDomain,
    pub submitted_at: i64,
}

#[event]
pub struct ProposalStatusChanged {
    pub proposal_id: Pubkey,
    pub new_status: ProposalStatus,
}

#[error_code]
pub enum ErrorCode {
    #[msg("Content CID cannot be zero")]
    InvalidContentCid,
    #[msg("Invalid status transition")]
    InvalidStatusTransition,
    #[msg("Unauthorized: only evaluator can update status")]
    Unauthorized,
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Proposer -> submit_proposal | Untrusted external input: title_hash, content_cid, repo_url_hash, demo_url_hash, domain |
| Evaluator -> update_status | Semi-trusted: any signer can call (Phase 6 adds proper authority) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-08 | Spoofing | submit_proposal | accept | Signer is the submitter identity via Solana's native signature verification |
| T-02-09 | Tampering | Proposal account | mitigate | No edit/close instructions exist. Account data set once in submit_proposal, only status field changes via update_status. PDA seeds prevent address collision. |
| T-02-10 | Denial of Service | submit_proposal | accept | Open submission (D-08). Rent cost (~0.00114 SOL) provides natural spam deterrent. Rate limiting deferred. |
| T-02-11 | Elevation of Privilege | update_status | accept | Currently any signer can call update_status. This is a known Phase 2 limitation -- Phase 6 will add evaluator authority config account with `has_one` constraint. Documented in code comment. |
| T-02-12 | Input Validation | content_cid | mitigate | require!(content_cid != [0u8; 32]) prevents empty content hash submission |
| T-02-13 | Tampering | PDA collision | mitigate | Seeds [b"proposal", submitter, title_hash] prevent same-submitter same-title duplicates. Different submitters or titles always produce different PDAs. |
</threat_model>
