    ```typescript
    import { getClientEnv } from "@/lib/env";
    import { type Address } from "viem";

    export function getContractAddresses(): { identityRegistry: Address; reputationRegistry: Address } {
      const env = getClientEnv();
      const identity = env.NEXT_PUBLIC_IDENTITY_REGISTRY_ADDRESS;
      const reputation = env.NEXT_PUBLIC_REPUTATION_REGISTRY_ADDRESS;
      if (!identity || !reputation) {
        throw new Error("Contract addresses not configured. Set NEXT_PUBLIC_IDENTITY_REGISTRY_ADDRESS and NEXT_PUBLIC_REPUTATION_REGISTRY_ADDRESS in .env.local");
      }
      return {
        identityRegistry: identity as Address,
        reputationRegistry: reputation as Address,
      };
    }
    ```
    NOTE: The `as Address` here is a viem-specific branded type narrowing. viem's Address type is `0x${string}`. Since env vars are validated strings, this is the standard viem pattern (not a type escape).

    Server-side only IPFS pinning using Pinata REST API (not SDK, to avoid extra dependency):
    ```typescript
    import { getServerEnv } from "@/lib/env";
    import type { PinataResponse } from "./types";

    const PINATA_API_URL = "https://api.pinata.cloud/pinning/pinJSONToIPFS";

    export async function pinJSON(data: unknown): Promise<string> {
      const env = getServerEnv();
      if (!env.PINATA_API_KEY || !env.PINATA_SECRET_KEY) {
        throw new Error("Pinata credentials not configured");
      }

      const response = await fetch(PINATA_API_URL, {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
          pinata_api_key: env.PINATA_API_KEY,
          pinata_secret_api_key: env.PINATA_SECRET_KEY,
        },
        body: JSON.stringify({ pinataContent: data }),
      });

      if (!response.ok) {
        const text = await response.text();
        throw new Error(`Pinata pin failed (${response.status}): ${text}`);
      }

      const result = (await response.json()) as PinataResponse;
      return result.IpfsHash;
    }

    export async function fetchFromIPFS<T>(cid: string): Promise<T> {
      const env = getServerEnv();
      const gatewayUrl = env.PINATA_GATEWAY_URL ?? "https://gateway.pinata.cloud/ipfs";
      const response = await fetch(`${gatewayUrl}/${cid}`);

      if (!response.ok) {
        throw new Error(`IPFS fetch failed for ${cid} (${response.status})`);
      }

      return response.json() as Promise<T>;
    }
    ```

    **src/app/api/proposals/submit/route.ts:**
    Create a POST endpoint that:
    1. Parses and validates request body with `proposalSchema` (Zod)
    2. Constructs `ProposalContent` object with `submittedAt: new Date().toISOString()`
    3. Pins content to IPFS via `pinJSON(content)` — returns CID
    4. Registers project identity on-chain: calls `IdentityRegistry.register(serverWalletAddress, ipfsCID)` via wallet client
    5. Waits for transaction receipt via public client `waitForTransactionReceipt`
    6. Extracts tokenId from `ProjectRegistered` event logs
    7. Returns JSON response: `{ tokenId: string, ipfsCID: string, txHash: string }`

    Error handling:
    - 400 for Zod validation failure (return `{ error: string, fieldErrors: Record<string, string[]> }`)
    - 500 for IPFS pin failure (return `{ error: "Failed to pin content to IPFS" }`)
    - 500 for chain write failure (return `{ error: "Failed to register on-chain" }`)
    - Use try/catch with specific error messages at each step

    ```typescript
    import { NextResponse } from "next/server";
    import { proposalSchema } from "@/lib/schemas/proposal";
    import { pinJSON } from "@/lib/ipfs/client";
    import { getPublicClient, getWalletClient } from "@/lib/chain/client";
    import { IDENTITY_REGISTRY_ABI, getContractAddresses } from "@/lib/chain/contracts";
    import type { ProposalContent } from "@/lib/ipfs/types";

    export async function POST(request: Request) {
      // 1. Validate input
      const body = await request.json();
      const parsed = proposalSchema.safeParse(body);
      if (!parsed.success) {
        return NextResponse.json(
          { error: "Validation failed", fieldErrors: parsed.error.flatten().fieldErrors },
          { status: 400 },
        );
      }

      // 2. Build IPFS content
      const content: ProposalContent = {
        ...parsed.data,
        submittedAt: new Date().toISOString(),
      };

      // 3. Pin to IPFS
      let ipfsCID: string;
      try {
        ipfsCID = await pinJSON(content);
      } catch (error) {
        return NextResponse.json(
          { error: "Failed to pin content to IPFS" },
          { status: 500 },
        );
      }

      // 4. Register on-chain
      try {
        const walletClient = getWalletClient();
        const publicClient = getPublicClient();
        const { identityRegistry } = getContractAddresses();

        const txHash = await walletClient.writeContract({
          address: identityRegistry,
          abi: IDENTITY_REGISTRY_ABI,
          functionName: "register",
          args: [walletClient.account.address, ipfsCID],
        });

        const receipt = await publicClient.waitForTransactionReceipt({ hash: txHash });

        // Extract tokenId from event logs
        // The ProjectRegistered event is the first topic
        const tokenId = receipt.logs[0]?.topics[1] ?? "0";

        return NextResponse.json({
          tokenId: String(BigInt(tokenId)),
          ipfsCID,
          txHash,
        });
      } catch (error) {
        return NextResponse.json(
          { error: "Failed to register on-chain" },
          { status: 500 },
        );
      }
    }
    ```

    Note: The tokenId extraction from logs may need adjustment based on actual event encoding. The executor should use viem's `decodeEventLog` with `IDENTITY_REGISTRY_ABI` and the `ProjectRegistered` event for reliable parsing.
