Create the Makefile for unified monorepo commands and the GitHub Actions CI workflow.

1. Create `Makefile` at project root:
```makefile
.PHONY: build test lint dev-solana dev-ethereum clean

build:
	anchor build
	cd contracts && forge build

test:
	bun test
	cd contracts && forge test -vvv

lint:
	cd contracts && forge fmt --check
	cd contracts && bunx solhint 'src/**/*.sol'
	cargo clippy --workspace -- -D warnings

fmt:
	cd contracts && forge fmt
	cargo fmt --all

dev-solana:
	solana-test-validator

dev-ethereum:
	anvil

clean:
	anchor clean
	cd contracts && forge clean
	rm -rf node_modules dist
```

IMPORTANT: Makefile recipes MUST use tabs (not spaces) for indentation. Each command line under a target must start with a tab character.

2. Create `.github/workflows/ci.yml`:
```yaml
name: CI

on:
  push:
    branches: [main]
  pull_request:
    branches: [main]

env:
  SOLANA_CLI_VERSION: "2.1.0"
  ANCHOR_VERSION: "1.0.0"
  NODE_VERSION: "23"

jobs:
  solana:
    name: Solana (Anchor)
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - uses: metadaoproject/setup-anchor@v3.3
        with:
          anchor-version: ${{ env.ANCHOR_VERSION }}
          solana-cli-version: ${{ env.SOLANA_CLI_VERSION }}
          node-version: ${{ env.NODE_VERSION }}

      - name: Generate keypair
        run: solana-keygen new --no-passphrase

      - name: Build Anchor programs
        run: anchor build

      - name: Clippy lint
        run: cargo clippy --workspace -- -D warnings

  ethereum:
    name: Ethereum (Foundry)
    runs-on: ubuntu-latest
    defaults:
      run:
        working-directory: ./contracts
    steps:
      - uses: actions/checkout@v4
        with:
          submodules: recursive

      - uses: foundry-rs/foundry-toolchain@v1

      - name: Build contracts
        run: forge build --sizes

      - name: Check formatting
        run: forge fmt --check

      - name: Run tests
        run: forge test -vvv

  lint:
    name: Lint
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - uses: oven-sh/setup-bun@v2

      - name: Install dependencies
        run: bun install

      - name: Solhint
        run: cd contracts && bunx solhint 'src/**/*.sol'
```

Verify the Makefile works locally:
```bash
make build
make test
make lint
```

Note: `make lint` may show warnings from clippy or solhint -- warnings are acceptable. Only errors should fail the build. The `cargo clippy -- -D warnings` flag treats warnings as errors intentionally to enforce clean code from the start.
