```typescript
import { z } from "zod";

export const RECOMMENDATIONS = ["strong_approve", "approve", "needs_revision", "reject"] as const;

export const evaluationOutputSchema = z.object({
  score: z.number().int().min(0).max(100),
  justification: z.string().min(1),
  recommendation: z.enum(RECOMMENDATIONS),
  keyFindings: z.array(z.string()).min(1).max(3),
});

export type EvaluationOutput = z.infer<typeof evaluationOutputSchema>;

// Named DIMENSION_KEYS (not DIMENSIONS) to avoid collision with the richer DIMENSIONS array in constants.ts
export const DIMENSION_KEYS = ["technical", "impact", "cost", "team"] as const;
export type EvaluationDimension = (typeof DIMENSION_KEYS)[number];

// Full evaluation result for one dimension including audit trail
export const dimensionEvaluationSchema = z.object({
  dimension: z.enum(DIMENSION_KEYS),
  output: evaluationOutputSchema,
  audit: z.object({
    promptSent: z.string(),
    modelUsed: z.string(),
    rawResponse: z.string(),
    evaluatedAt: z.number(), // Unix timestamp ms
  }),
});

export type DimensionEvaluation = z.infer<typeof dimensionEvaluationSchema>;

// Complete proposal evaluation (all 4 dimensions + aggregate)
export const proposalEvaluationSchema = z.object({
  proposalId: z.string(),
  dimensions: z.array(dimensionEvaluationSchema).min(1).max(4),
  aggregate: z.object({
    weightedScore: z.number().min(0).max(100),
    completedDimensions: z.number().int().min(1).max(4),
    computedAt: z.number(),
  }),
  status: z.enum(["evaluating", "evaluated", "failed"]),
});

export type ProposalEvaluation = z.infer<typeof proposalEvaluationSchema>;
```

import type { EvaluationDimension } from "./schemas";

export const DIMENSION_WEIGHTS: Record<EvaluationDimension, number> = {
  technical: 0.25,
  impact: 0.30,
  cost: 0.20,
  team: 0.25,
} as const;

export const SCORE_BANDS = {
  exceptional: { min: 81, max: 100, label: "Exceptional" },
  strong: { min: 61, max: 80, label: "Strong" },
  adequate: { min: 41, max: 60, label: "Adequate" },
  weak: { min: 21, max: 40, label: "Weak" },
  insufficient: { min: 0, max: 20, label: "Insufficient" },
} as const;

// Rich dimension metadata array for UI rendering and orchestration.
// This is the canonical DIMENSIONS export -- schemas.ts exports DIMENSION_KEYS (string tuple only).
export const DIMENSIONS = [
  { key: "technical" as const, label: "Technical Feasibility", weight: 0.25 },
  { key: "impact" as const, label: "Impact Potential", weight: 0.30 },
  { key: "cost" as const, label: "Cost Efficiency", weight: 0.20 },
  { key: "team" as const, label: "Team Capability", weight: 0.25 },
] as const;

export const IPE_CITY_VALUES = `IPE City core values that should inform your evaluation:
1. Pro-technological innovation: Favor proposals that advance technology and push boundaries
2. Pro-freedom: Favor proposals that increase individual autonomy, decentralization, and open systems
3. Pro-human progress: Favor proposals that measurably improve human capability and quality of life`;

export const MODEL_CONFIG = {
  model: "gpt-4o" as const,
  temperature: 0.3,
  maxTokens: 1500,
} as const;

/**
 * Derive score band label from SCORE_BANDS object.
 * Iterates bands ordered by min descending so the first match wins.
 */
export function getScoreBand(score: number): string {
  const bands = Object.values(SCORE_BANDS);
  for (const band of bands) {
    if (score >= band.min && score <= band.max) {
      return band.label;
    }
  }
  return "Insufficient";
}

**prompts.ts:**

Define `DIMENSION_RUBRICS` -- an object keyed by dimension with dimension-specific rubric text for each of the 5 bands. Each dimension rubric describes what scores 81-100, 61-80, 41-60, 21-40, 0-20 mean for THAT specific dimension.

Technical Feasibility rubric:
- 81-100: Proven tech stack, clear architecture with diagrams/specs, realistic timeline with contingency buffer, all technical risks identified and mitigated
- 61-80: Solid technical approach with minor gaps in architecture or timeline
- 41-60: Feasible but with significant technical risks or unclear implementation details
- 21-40: Major technical challenges unaddressed, unclear how this would be built
- 0-20: Technically infeasible or no technical detail provided

Impact Potential rubric:
- 81-100: Clear measurable impact metrics, affects large population, directly advances IPE City mission, novel approach to a real problem
- 61-80: Good impact potential with some metrics, meaningful scope, aligns with mission
- 41-60: Moderate impact, limited scope or unclear measurement plan
- 21-40: Marginal impact, narrow audience, weak connection to mission
- 0-20: No discernible impact or does not align with IPE City values

Cost Efficiency rubric:
- 81-100: Detailed budget with line items, excellent cost-to-impact ratio, sustainable funding model, clear milestones tied to deliverables
- 61-80: Reasonable budget with minor gaps, good value proposition
- 41-60: Budget exists but lacks detail, some concerns about cost-to-impact ratio
- 21-40: Vague budget, poor cost justification, high risk of overspend
- 0-20: No budget information or clearly unreasonable costs

Team Capability rubric:
- 81-100: Proven track record in relevant domain, complementary skills, clear role assignments, evidence of prior successful projects
- 61-80: Competent team with relevant experience, minor gaps in coverage
- 41-60: Team has some relevant experience but significant skill gaps
- 21-40: Limited relevant experience, unclear team composition
- 0-20: No team information or clearly inadequate team for proposed scope

`buildSystemPrompt(dimension: EvaluationDimension): string` constructs the full system prompt by combining:
1. Role statement: "You are a {label} Judge for IPE City grants."
2. Isolation instruction: "Evaluate ONLY the {label} of the proposal. Do NOT consider other dimensions -- those are evaluated by other independent judges."
3. The dimension-specific rubric (from DIMENSION_RUBRICS)
4. Scoring guidance: "Use the full 0-100 range. A score of 50 means adequate. Most proposals should score between 30-70. Scores above 80 require exceptional merit with no significant gaps."
5. IPE City values (from constants.ts)
6. Final instruction: "Evaluate the following proposal:"

`NAIVE_PROMPT` = "Evaluate this grant proposal and give it a score from 0 to 100. Provide a justification and key findings."

**`evaluateDimension(dimension: EvaluationDimension, proposalText: string): Promise<DimensionEvaluation>`**
1. Build the system prompt: `const systemPrompt = buildSystemPrompt(dimension);`
2. Call OpenAI: `const completion = await openai.chat.completions.parse({ model: MODEL_CONFIG.model, messages: [{ role: "system", content: systemPrompt }, { role: "user", content: proposalText }], response_format: zodResponseFormat(evaluationOutputSchema, "evaluation"), temperature: MODEL_CONFIG.temperature, max_tokens: MODEL_CONFIG.maxTokens });`
3. Check for refusal: `if (completion.choices[0]?.message.refusal) { throw new Error(\`OpenAI refused evaluation for \${dimension}: \${completion.choices[0].message.refusal}\`); }`
4. Extract parsed: `const parsed = completion.choices[0]?.message.parsed;` -- if null, throw Error with "Failed to parse structured output for {dimension}".
5. Return DimensionEvaluation: `{ dimension, output: parsed, audit: { promptSent: systemPrompt, modelUsed: completion.model, rawResponse: JSON.stringify(completion.choices[0]?.message), evaluatedAt: Date.now() } }`

**`evaluateAllDimensions(proposalText: string): Promise<DimensionEvaluation[]>`**
1. `const results = await Promise.all(DIMENSION_KEYS.map(dim => evaluateDimension(dim, proposalText)));`
2. Return results array.

**`evaluateNaive(proposalText: string): Promise<string>`**
1. Call OpenAI WITHOUT structured output: `const completion = await openai.chat.completions.create({ model: MODEL_CONFIG.model, messages: [{ role: "system", content: NAIVE_PROMPT }, { role: "user", content: proposalText }], temperature: MODEL_CONFIG.temperature, max_tokens: MODEL_CONFIG.maxTokens });`
2. Return `completion.choices[0]?.message.content ?? ""`.

The OpenAI client is instantiated at module level: `const openai = new OpenAI();` -- it reads `OPENAI_API_KEY` from env.

| Boundary | Description |
|----------|-------------|
| Proposal text -> OpenAI prompt | Untrusted user-submitted proposal text is injected into LLM system prompts |
| OpenAI response -> application | LLM output is parsed and stored; refusal or malformed output possible |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-01 | Tampering | agents.ts (prompt injection) | mitigate | Proposal text is always passed as the `user` message, never concatenated into the `system` message. System prompt is a fixed template. OpenAI structured output mode constrains output format, preventing the model from outputting arbitrary content. |
| T-02-02 | Information Disclosure | agents.ts (API key) | mitigate | OpenAI client reads OPENAI_API_KEY from server-side env only. agents.ts runs server-side in Next.js API routes, never in client bundles. |
| T-02-03 | Denial of Service | agents.ts (cost abuse) | accept | Rate limiting is handled at the API route level (Plan 02), not in the agent functions themselves. Agent functions are internal library code. |
</threat_model>
