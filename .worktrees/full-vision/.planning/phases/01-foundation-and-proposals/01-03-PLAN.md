    **src/lib/ipfs/client.ts:**
    Server-side only IPFS pinning using Pinata REST API. Use Zod to parse the response instead of `as PinataResponse`. Use `getRequiredServerEnv()` for fail-fast behavior:
    ```typescript
    import { z } from "zod";
    import { getRequiredServerEnv } from "@/lib/env";
    import { pinataResponseSchema } from "./types";

    const PINATA_API_URL = "https://api.pinata.cloud/pinning/pinJSONToIPFS";

    export async function pinJSON(data: unknown): Promise<string> {
      const env = getRequiredServerEnv();

      const response = await fetch(PINATA_API_URL, {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
          pinata_api_key: env.PINATA_API_KEY,
          pinata_secret_api_key: env.PINATA_SECRET_KEY,
        },
        body: JSON.stringify({ pinataContent: data }),
      });

      if (!response.ok) {
        const text = await response.text();
        throw new Error(`Pinata pin failed (${response.status}): ${text}`);
      }

      const json: unknown = await response.json();
      const result = pinataResponseSchema.parse(json);
      return result.IpfsHash;
    }

    export async function fetchFromIPFS<T>(cid: string, schema: z.ZodType<T>): Promise<T> {
      const env = getRequiredServerEnv();
      const response = await fetch(`${env.PINATA_GATEWAY_URL}/${cid}`);

      if (!response.ok) {
        throw new Error(`IPFS fetch failed for ${cid} (${response.status})`);
      }

      const json: unknown = await response.json();
      return schema.parse(json);
    }
    ```
    NOTE: `fetchFromIPFS` takes a Zod schema parameter to validate the response, eliminating the need for generic type assertions. Callers must provide the expected schema (e.g., `proposalContentSchema`).
    NOTE: Both functions now use `getRequiredServerEnv()` instead of `getServerEnv()` -- missing Pinata credentials or RPC URL will throw immediately with a descriptive error instead of silently returning undefined (addresses review concern #5).

    **src/app/api/proposals/submit/route.ts:**
    Create a POST endpoint that:
    1. Parses and validates request body with `proposalSchema` (Zod)
    2. Constructs `ProposalContent` object with `submittedAt: new Date().toISOString()`
    3. Pins content to IPFS via `pinJSON(content)` -- returns CID
    4. Registers project identity on-chain: calls `IdentityRegistry.register(serverWalletAddress, ipfsCID)` via wallet client
    5. Waits for transaction receipt via public client `waitForTransactionReceipt`
    6. Extracts tokenId from `ProjectRegistered` event logs using viem's `decodeEventLog`
    7. Returns JSON response: `{ tokenId: string, ipfsCID: string, txHash: string }`

    **Addresses review concern #3 (server-owns-all identity model):** Add a code comment at the `register()` call site documenting that in v1, the server's deployer wallet owns all project identities because there is no user authentication. When auth is added (v2), ownership should transfer to the authenticated user's address.

    **Addresses review concern #2 (orphaned IPFS pins):** If the on-chain registration fails AFTER IPFS pin succeeds, the error response MUST include the orphaned `ipfsCID` so the user/operator can either retry the chain write or clean up the pin. Return `{ error: "...", ipfsCID: "Qm..." }` in the 500 response.

    **Addresses review concern #9 (tokenId falls back to "0"):** If `decodeEventLog` does not find a `ProjectRegistered` event in the receipt logs, throw an explicit error instead of silently defaulting to tokenId "0". This indicates a contract bug or ABI mismatch that should fail loudly.

    ```typescript
    import { NextResponse } from "next/server";
    import { decodeEventLog } from "viem";
    import { proposalSchema } from "@/lib/schemas/proposal";
    import { pinJSON } from "@/lib/ipfs/client";
    import { getPublicClient, getWalletClient } from "@/lib/chain/client";
    import { IDENTITY_REGISTRY_ABI, getContractAddresses } from "@/lib/chain/contracts";
    import type { ProposalContent } from "@/lib/ipfs/types";

    export async function POST(request: Request) {
      // 1. Validate input
      const body: unknown = await request.json();
      const parsed = proposalSchema.safeParse(body);
      if (!parsed.success) {
        return NextResponse.json(
          { error: "Validation failed", fieldErrors: parsed.error.flatten().fieldErrors },
          { status: 400 },
        );
      }

      // 2. Build IPFS content
      const content: ProposalContent = {
        ...parsed.data,
        submittedAt: new Date().toISOString(),
      };

      // 3. Pin to IPFS
      let ipfsCID: string;
      try {
        ipfsCID = await pinJSON(content);
      } catch {
        return NextResponse.json(
          { error: "Failed to pin content to IPFS" },
          { status: 500 },
        );
      }

      // 4. Register on-chain
      // NOTE (v1 design decision): The server's deployer wallet owns all project
      // identities because there is no user authentication in v1 (UI-04: public access).
      // When auth is added (AUTH-01/AUTH-02 in v2), the owner should be the
      // authenticated user's wallet address instead of walletClient.account.address.
      try {
        const walletClient = getWalletClient();
        const publicClient = getPublicClient();
        const { identityRegistry } = getContractAddresses();

        const txHash = await walletClient.writeContract({
          address: identityRegistry,
          abi: IDENTITY_REGISTRY_ABI,
          functionName: "register",
          args: [walletClient.account.address, ipfsCID],
        });

        const receipt = await publicClient.waitForTransactionReceipt({ hash: txHash });

        // Extract tokenId from ProjectRegistered event using viem decodeEventLog.
        // This reliably decodes the event instead of raw topics parsing.
        // Addresses review concern #9: throw if event not found instead of defaulting to "0".
        let tokenId: string | undefined;
        for (const log of receipt.logs) {
          try {
            const decoded = decodeEventLog({
              abi: IDENTITY_REGISTRY_ABI,
              data: log.data,
              topics: log.topics,
            });
            if (decoded.eventName === "ProjectRegistered") {
              tokenId = String(decoded.args.tokenId);
              break;
            }
          } catch {
            // Not a matching event log, continue to next
          }
        }

        if (!tokenId) {
          // This should never happen if the contract and ABI are correct.
          // Failing loudly here catches ABI mismatches or contract bugs early.
          return NextResponse.json(
            {
              error: "Chain registration succeeded but ProjectRegistered event not found in receipt. This indicates an ABI mismatch.",
              ipfsCID,
              txHash,
            },
            { status: 500 },
          );
        }

        return NextResponse.json({
          tokenId,
          ipfsCID,
          txHash,
        });
      } catch {
        // Addresses review concern #2: return the orphaned IPFS CID so the
        // user/operator can retry the chain write or clean up the pin.
        return NextResponse.json(
          { error: "Failed to register on-chain. Content was pinned to IPFS but not registered.", ipfsCID },
          { status: 500 },
        );
      }
    }
    ```
