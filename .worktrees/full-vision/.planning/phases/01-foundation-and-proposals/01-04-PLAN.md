    **src/app/api/proposals/route.ts (GET -- list all proposals):**

    This is the "rebuildable from chain events" approach (STORE-04). Several review concerns are addressed here:

    **Addresses review concern #1 (unbounded chain scan from block 0):** Use `DEPLOYMENT_BLOCK_NUMBER` from env as `fromBlock` instead of `0n`. This is the block where contracts were deployed (captured in Plan 01 Task 3). Scanning from genesis on a growing chain gets progressively more expensive -- using the deployment block scans only the relevant range.

    **Addresses review concern #6 (no caching):** Add `Cache-Control: public, s-maxage=30, stale-while-revalidate=60` response header. This allows Vercel's edge cache to serve stale data for 30 seconds and revalidate in background for another 60 seconds, dramatically reducing chain+IPFS load.

    **Addresses review concern #7 (proposalContentSchema inline):** Import `proposalContentSchema` from `@/lib/ipfs/types` (where Plan 03 also imports it) instead of redefining it inline. Single source of truth for the IPFS content shape.

    **Addresses review concern #1 (N unbounded IPFS fetches):** Use a concurrency-limited fetch pattern instead of unbounded `Promise.all`. Limit to 5 concurrent IPFS fetches to avoid gateway rate limiting.

    ```typescript
    import { NextResponse } from "next/server";
    import { getPublicClient } from "@/lib/chain/client";
    import {
      IDENTITY_REGISTRY_ABI,
      REPUTATION_REGISTRY_ABI,
      getContractAddresses,
    } from "@/lib/chain/contracts";
    import { fetchFromIPFS } from "@/lib/ipfs/client";
    import { proposalContentSchema } from "@/lib/ipfs/types";
    import { getRequiredServerEnv } from "@/lib/env";

    // Concurrency-limited Promise.all -- processes items in batches of `limit`
    async function mapWithConcurrency<T, R>(
      items: T[],
      limit: number,
      fn: (item: T) => Promise<R>,
    ): Promise<R[]> {
      const results: R[] = [];
      for (let i = 0; i < items.length; i += limit) {
        const batch = items.slice(i, i + limit);
        const batchResults = await Promise.all(batch.map(fn));
        results.push(...batchResults);
      }
      return results;
    }

    const IPFS_CONCURRENCY_LIMIT = 5;

    export async function GET() {
      try {
        const publicClient = getPublicClient();
        const env = getRequiredServerEnv();
        const { identityRegistry, reputationRegistry } = getContractAddresses();

        // Use deployment block as fromBlock -- NOT 0n (review concern #1)
        const deploymentBlock = BigInt(env.DEPLOYMENT_BLOCK_NUMBER);

        // Fetch all ProjectRegistered events from deployment block onward
        const logs = await publicClient.getLogs({
          address: identityRegistry,
          event: {
            type: "event",
            name: "ProjectRegistered",
            inputs: [
              { name: "tokenId", type: "uint256", indexed: true },
              { name: "owner", type: "address", indexed: true },
              { name: "agentURI", type: "string", indexed: false },
            ],
          },
          fromBlock: deploymentBlock,
          toBlock: "latest",
        });

        // Enrich each proposal with reputation data and IPFS content
        // Use concurrency-limited fetches to avoid IPFS gateway rate limiting (review concern #1)
        const proposals = await mapWithConcurrency(
          logs,
          IPFS_CONCURRENCY_LIMIT,
          async (log) => {
            const tokenId = log.args.tokenId?.toString() ?? "0";
            const owner = log.args.owner ?? "0x0";
            const ipfsCID = log.args.agentURI ?? "";

            // Get reputation summary
            const [feedbackCount, averageScore] = await publicClient.readContract({
              address: reputationRegistry,
              abi: REPUTATION_REGISTRY_ABI,
              functionName: "getSummary",
              args: [BigInt(tokenId)],
            });

            // Determine status from feedback count
            const count = Number(feedbackCount);
            const status = count > 0 ? "evaluated" : "submitted";

            // Fetch IPFS content for display -- graceful degradation on failure
            let title = "Content unavailable";
            let description = "";
            let budget = 0;
            let submittedAt = "";
            try {
              const content = await fetchFromIPFS(ipfsCID, proposalContentSchema);
              title = content.title;
              description = content.description;
              budget = content.budget;
              submittedAt = content.submittedAt;
            } catch {
              // IPFS gateway may be temporarily unavailable -- show partial data
            }

            return {
              tokenId,
              owner,
              ipfsCID,
              status,
              feedbackCount: count,
              averageScore: Number(averageScore),
              title,
              description,
              budget,
              submittedAt,
            };
          },
        );

        // Sort newest first (proposals with failed IPFS fetch have empty submittedAt, sort to end)
        proposals.sort((a, b) => {
          if (!a.submittedAt && !b.submittedAt) return 0;
          if (!a.submittedAt) return 1;
          if (!b.submittedAt) return -1;
          return b.submittedAt.localeCompare(a.submittedAt);
        });

        // Cache response for 30s, allow stale for 60s more (review concern #6)
        return NextResponse.json(proposals, {
          headers: {
            "Cache-Control": "public, s-maxage=30, stale-while-revalidate=60",
          },
        });
      } catch {
        return NextResponse.json(
          { error: "Failed to fetch proposals" },
          { status: 500 },
        );
      }
    }
    ```

    **src/app/api/proposals/[tokenId]/route.ts (GET -- single proposal detail):**

    **Addresses review concern #8 (useParams returns string | string[]):** The API route receives params server-side via Next.js route handler convention. Validate that tokenId is a single string and a valid positive integer before using it.

    1. Extract `tokenId` from route params
    2. Validate tokenId is a non-empty string representing a positive integer (not an array, not NaN)
    3. Call `IdentityRegistry.getMetadata(tokenId)` to get `owner` and `agentURI` (IPFS CID)
    4. Fetch full content from IPFS via `fetchFromIPFS(agentURI, proposalContentSchema)` -- import from `@/lib/ipfs/types`
    5. Call `ReputationRegistry.getSummary(tokenId)` for reputation data
    6. Return full proposal detail:
       ```typescript
       {
         tokenId: string;
         owner: string;
         ipfsCID: string;
         status: "submitted" | "evaluating" | "evaluated";
         feedbackCount: number;
         averageScore: number;
         content: ProposalContent; // full IPFS content
       }
       ```
    7. Return 400 if tokenId param is invalid (not a positive integer)
    8. Return 404 if contract call reverts (token doesn't exist)
    9. Return 502 if IPFS fetch fails (gateway unavailable)
    10. Add `Cache-Control: public, s-maxage=60, stale-while-revalidate=120` -- detail pages change less frequently than list

    ```typescript
    import { NextResponse } from "next/server";
    import { z } from "zod";
    import { getPublicClient } from "@/lib/chain/client";
    import {
      IDENTITY_REGISTRY_ABI,
      REPUTATION_REGISTRY_ABI,
      getContractAddresses,
    } from "@/lib/chain/contracts";
    import { fetchFromIPFS } from "@/lib/ipfs/client";
    import { proposalContentSchema } from "@/lib/ipfs/types";

    // Validates tokenId is a string representing a positive integer
    const tokenIdSchema = z.string().regex(/^\d+$/, "tokenId must be a positive integer").transform(Number).pipe(z.number().positive());

    export async function GET(
      _request: Request,
      { params }: { params: Promise<{ tokenId: string }> },
    ) {
      const { tokenId: rawTokenId } = await params;

      // Validate tokenId param (review concern #8)
      const parsedTokenId = tokenIdSchema.safeParse(rawTokenId);
      if (!parsedTokenId.success) {
        return NextResponse.json(
          { error: "Invalid proposal ID. Must be a positive integer." },
          { status: 400 },
        );
      }

      try {
        const publicClient = getPublicClient();
        const { identityRegistry, reputationRegistry } = getContractAddresses();
        const tokenIdBigInt = BigInt(parsedTokenId.data);

        // Get on-chain metadata
        const [owner, agentURI] = await publicClient.readContract({
          address: identityRegistry,
          abi: IDENTITY_REGISTRY_ABI,
          functionName: "getMetadata",
          args: [tokenIdBigInt],
        });

        // Get reputation summary
        const [feedbackCount, averageScore] = await publicClient.readContract({
          address: reputationRegistry,
          abi: REPUTATION_REGISTRY_ABI,
          functionName: "getSummary",
          args: [tokenIdBigInt],
        });

        const count = Number(feedbackCount);
        const status = count > 0 ? "evaluated" : "submitted";

        // Fetch full content from IPFS
        let content;
        try {
          content = await fetchFromIPFS(agentURI, proposalContentSchema);
        } catch {
          return NextResponse.json(
            { error: "Unable to load proposal content. The IPFS gateway may be temporarily unavailable." },
            { status: 502 },
          );
        }

        return NextResponse.json(
          {
            tokenId: rawTokenId,
            owner,
            ipfsCID: agentURI,
            status,
            feedbackCount: count,
            averageScore: Number(averageScore),
            content,
          },
          {
            headers: {
              "Cache-Control": "public, s-maxage=60, stale-while-revalidate=120",
            },
          },
        );
      } catch {
        // Contract call reverts for non-existent tokens
        return NextResponse.json(
          { error: "Proposal not found" },
          { status: 404 },
        );
      }
    }
    ```
